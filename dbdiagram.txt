Table venues {
  id integer
  name string
  address string
  size integer
  maximumOccupancy integer
  image url
}

Table bands {
  id integer
  name string
  members integer
  genre string
  yearFounded integer
  image url
}

Table bookings {
  bandId integer
  venueId integer
  date integer
}

Ref: "bookings"."bandId" < "bands"."id"

Ref: "bookings"."venueId" < "venues"."id"
